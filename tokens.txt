Se incontro un CHAR
	i token validi sono:
		'&', '|', '<', '>', '(', ')' char

Se incontro un '&'
	i token validi sono:
		'<', '>', '(', char

Se incontro un '|'
	i token validi sono:
		'<', '>', '(', char

Se incontro un '<'
	i token validi sono:
		char, '&', '|'

Se incotntro un '<<'
	i token validi sono:
		char, '&', '|'

Se incontro un '('
	i token validi sono:
		'>', '<', char

Se incontro un ')'
	i token validi sono:
		'&', '|', '<', '>'
