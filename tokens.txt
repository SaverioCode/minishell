Se incontro un CHAR
	i token validi sono:
		'&', '|', '<', '>', '('

Se incontro un '&'
	i token validi sono:
		'<', '>', char

Se incontro un '|'
	i token validi sono:
		'<', '>', char

Se incontro un '<'
	i token validi sono:
		'&', '|', char

Se incontro un '('
	i token validi sono:
		'>', '<', char

Se incontro un ')'
	i token validi sono:
		')', '&', '|'
